\item (SQL seulement) Quel est le nom du chauffeur (ou des chauffeurs) de Paris qui a fait le plus de courses~? (2 points)\\
\begin{correction}
\begin{verbatim}

FROM COURSES, CHAUFFEURS

\item (AlgŤbre seulement) Donnez le id du client (ou des clients) qui a effectuť la course la plus chŤre~? (1 point)\\

MMAX =$\pi_{montantFacture}(COURSES) - \pi_{montantFacture}(COURSES \Join_{montantFacture < montantFacture'} COURSES')$\\

\item (SQL) Quels sont les id des clients de Tours qui ont donnť une note  globale
moyenne infťrieure ŗ 10 aux chauffeurs qu'ils ont notťs~? Triez ces clients 
en ordre croissant de la moyenne de la note globale. (2 points)\\
\begin{correction}
\begin{smallverbatim}
SELECT cl.nom
FROM CLIENT cl, COURSES co
WHERE cl.id = co.id_client AND cl.ville = 'Tours'
GROUP BY id_client
HAVING AVG(noteGlobale) < 10
ORDER BY AVG(noteGlobale);
\end{smallverbatim}
\end{correction}

\item (AlgŤbre) Donnez les id des clients qui ont effectuť au moins une course avec chaque chauffeur~?  (1 points)\\
\begin{correction}
$A = \pi_{idClient, idChauffeur}(CHAUFFEURS)$ \\
$B = \rho_{id\rightarrow idChauffeur}(\pi_{id}(CHAUFFEURS))$\\
$Resultat = A \div B$
\end{correction}

\item (AlgŤbre + SQL) Donnez les id de chauffeurs qui n'ont transportť que des clients plus ‚gťs que eux-mÍmes.  (2 points)\\

\begin{correction}
$CH = \rho_{id\rightarrow idChauffeur, age\rightarrow ageChauffeur}(CHAUFFEURS)$\\
$CL = \rho_{id\rightarrow idClient, age\rightarrow ageClient}(CLIENTS)$\\
$IDCH = \rho_{id\rightarrow idChauffeur}(\pi_{id}(CHAUFFEURS))$\\
$Res = IDCH - \pi_{idChauffeur}(\sigma_{ageChauffeur >= ageClient}(CL \Join (COURSE \Join CH)))$
\begin{smallverbatim}
SELECT id FROM CHAUFFEURS
WHERE id NOT IN (SELECT ch.id
                 FROM CHAUFFEURS ch, COURSES co, CLIENTS cl
                 WHERE ch.id = co.idChauffeur AND cl.id = co.idClient 
                 AND ch.age >= cl.age)
\end{smallverbatim}
\end{correction}

\end{enumerate}
