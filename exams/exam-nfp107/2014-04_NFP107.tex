\section{Algèbre relationnelle et SQL (10 points)}

Soit l'extrait suivant de la base de données qui sert à gérer un parc d'avions :

\smallskip
\texttt{\textbf{VOYAGE}(numVol, villeDepart, villeArrivée, dateDepart, dateArrivée)}\\
\texttt{\textbf{VOL}(numVol, compagnie, numAvion, nombreVoyageurs)}\\
\texttt{\textbf{AVION}(numAvion, modèle, capacité, consommation, reserveCombustible)}\\
\texttt{\textbf{DISTANCE}(ville1, ville2, dist)}\\
\texttt{\textbf{VILLE}(nomVille, population, continent)}\\

\smallskip

\begin{enumerate}

\item Pour la table DISTANCE, identifiez les clés primaires et étrangères. (0,5 points)
\begin{correction}
Clé primaire: (ville1, ville2)\\
Clés étrangères: ville1,ville2 $\rightarrow$ VILLE(nomVille)
\end{correction}

\item Donnez la commande SQL pour créer la table DISTANCE. (0,5 points)
\begin{correction}
\begin{verbatim}
CREATE TABLE DISTANCE(
    ville1 VARCHAR2(10),
    ville2 VARCHAR2(10),
    population NUMBER(10),
    continent VARCHAR2(10),
    CONSTRAINT pk_distance PRIMARY_KEY (ville1, ville2)
    CONSTRAINT fk1_distance FOREIGN_KEY ville1 REFERENCES VILLE(nomVille),
    CONSTRAINT fk2_distance FOREIGN_KEY ville2 REFERENCES VILLE(nomVille))
\end{verbatim}
\end{correction}

\item (Algèbre et SQL) Donnez les vols qui durent plus de 10 heures.(1 point)
\begin{correction}

$\sigma_{dateArrivée - dateDepart > 10 \; heures}(VOYAGE)$

\begin{verbatim}
SELECT * FROM VOYAGE
WHERE (dateArrivée - dateDepart)*24 > 10;
\end{verbatim}
\end{correction}

\item (Algèbre et SQL) Les voyages qui embarquent plus de 400 voyageurs. (1 point)
\begin{correction}

$\sigma_{nombreVoyageurs > 400}(VOYAGE \Join VOL)$

\begin{verbatim}
SELECT * FROM VOYAGE voy, VOL v
WHERE (voy.numVol=v.numVol AND nombreVoyageurs > 400;
\end{verbatim}
\end{correction}

\item (Algèbre et SQL seulement) Nombre de vols au départ de Paris vers l'Afrique qui embarquent moins de 50\% de leur capacité (3 points):
\begin{correction}

\begin{multline}
\sigma_{villeDepart='Paris' \land continent='Afrique' \land nbVoyageurs < capacite/2}\\ (((VOYAGE \Join VOL) \Join AVION) \Join_{villeArrivee=nomVille} VILLE)
\end{multline}

\begin{verbatim}

SELECT COUNT(voy.numVol)
FROM VOYAGE voy, VOL v, AVION a, VILLE vl
WHERE voy.numVol = v.numVol AND v.numAvion = a.numAvion AND 
      villeDepart = 'Paris' AND villeArrivee = vl.nomVille AND 
      v.continent = 'Afrique' AND nbVoyageurs < capacite/2;
\end{verbatim}
\end{correction}

\item (SQL seulement) Le modèle d'avion qui transporte le plus de voyageurs (tous vols compris) (2 points).

\begin{correction}
\begin{verbatim}
SELECT modele
FROM VOL v, AVION a
WHERE v.numAvion = a.numAvion
GROUP BY modele
HAVING SUM(nombreVoyageurs) = (SELECT MAX(SUM(nombreVoyageurs))
                               FROM VOL v, AVION a
                               WHERE v.numAvion = a.numAvion
                               GROUP BY modele)

\end{verbatim}
\end{correction}

\item (Algèbre seulement) Quelle est la  (ou les) ville(s) destination vers
laquelle toutes les compagnies ont des vols ? (1 point).

\begin{correction}
$\pi_{villeArrivee, compagnie}(VOYAGE \Join VOL) \div \pi_{compagnie}(VOL)$
\end{correction}

\item (Algèbre seulement) Quels sont les modèles d'avion qui consomment moins
que tous les avions de la compagnie AIR FRANCE? (1 point).

\begin{correction}
\begin{multline}
\pi_{modele}(AVION) - \\
\pi_{modele}(AVION \Join_{consommation > consommation'}\rho_{consommation/consommation'}(\\
\pi_{consommation}(\sigma_{compagnie = 'AIR FRANCE'}(VOL) \Join AVION)))
\end{multline}

\end{correction}


\end{enumerate}

\section{Optimisation (4 points)}

	\begin{verbatim}
	SELECT DISTINCT villeArrivée
	FROM Vol V, Voyage Vo
	WHERE V.Compagnie = 'Air France' AND
	    V.NumVol = Vo.NumVol AND
	    Vo.VilleDépart = 'Paris';
	\end{verbatim}
\begin{enumerate}
	\item Donner en une phrase ce que produit cette requête ({\tt 1 point}).
\begin{correction}
Cette requête donne les destinations distinctes des vols au départ de Paris pour
la compagnie 'Air France'.
\end{correction}
	\item Donner \underline{deux} plans d'exécution EXPLAIN différents à partir de
	cette requête. Sachant que~: ({\tt 3 points})
	\begin{itemize}
	  \item l'opération pour le distinct est un {\it 'SORT UNIQUE'}, juste après le

		\end{itemize}
	 \end{itemize}

\end{enumerate}

H = $r_1[x] r_2[x] r_3[y] w_1[x] w_1[z] c_1 w_2[y] c_2 w_3[z] c_3$

\begin{enumerate}

\item Donner la liste des conflits de H; \texttt{(0,5 point)}
\begin{correction}
	\begin{itemize}
		\item[sur x : ] $r_2[x] w_1[x]$
		\item[sur y : ] $r_3[y] w_2[y]$
		\item[sur z : ] $w_1[z] w_3[z]$
	\end{itemize}
\end{correction}

\item Donner le graphe de s\'erialisation de H. Que pouvez-vous déduire de ce
graphe? \texttt{(0,5 point)}
\begin{correction}
	$T_2 \rightarrow T_1, T_3 \rightarrow T_2, T_1 \rightarrow T_3$. Il y a un cycle, dont H n'est pas \textit{s\'erialisable}.
\end{correction}
