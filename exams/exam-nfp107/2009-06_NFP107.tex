\section{Algčbre et SQL (8 points)}

Soit le schéma de base de données suivant :
\begin{itemize}
	\item Bateau (\underline{Num\_Bat}, NomB, Sponsor) ;
	\item Course (\underline{Num\_Cou}, NomC, Date, Prix) ;
	\item Resultat (\underline{Num\_Bat, Num\_Cou}, Score) ;
\end{itemize}

\begin{enumerate}
\item[] \textbf{Donner une expression algébrique et une requęte en SQL pour les phrases suivantes :}\\
\item Donner les bateaux sponsorisés par 'Chambourcy' \texttt{(0.5 point)};
\begin{correction}
	$\sigma_{Sponsor='Chambourcy'}Bateau$\\
	SELECT * FROM Bateau WHERE Sponsor='Chambourcy' ;
\end{correction}

\item Donner le nom des courses contenant le mot 'solitaire' et dont le prix est supérieur ŕ 10 000 euros \texttt{(0.5 point)}
\begin{correction}
	$\Pi_{NomC}(\sigma_{NomC='\%solitaire\%' \land Prix > 10000}Course)$\\
	SELECT NomC FROM Course Where NomC like '\%solitaire\%' AND Prix > 10000 ;
\end{correction}

\item Donner le nom des bateaux (distincts) qui sont arrivés premiers au moins une fois.\\Ecrire en SQL de 2 maničres différentes \texttt{(1.5 points)}
\begin{correction}
	$\Pi_{NomB}(\sigma_{Score=1}Course \Join Bateau)$\\
	SELECT DISTINCT NomB FROM Bateau B, Resultat R\\\hspace*{2.5cm}WHERE B.Num\_Bat=R.Num\_Bat AND Score=1 ;\\
	SELECT DISTINCT NomB FROM Bateau B WHERE B.Num\_Bat IN\\\hspace*{2.5cm}(SELECT R.Num\_Bat FROM Resultat WHERE Score=1) ;\\
	SELECT DISTINCT NomB FROM Bateau B WHERE EXISTS\\\hspace*{2.5cm}(SELECT R.Num\_Bat FROM Resultat R WHERE R.Num\_Bat=B.Num\_Bat AND Score=1) ;\\
\end{correction}

\item Donner la liste des noms de bateaux ayant participé au 'Solo de Quiberon' \texttt{(1 point)} ;
\begin{correction}
	$\Pi_{NomB}(Bateau \Join Resultat \Join \sigma_{NomC='Solo de Quiberon'}Course)$\\
	SELECT NomB FROM Bateau B, Resultat R, Course C\\\hspace*{2.5cm}WHERE B.Num\_Bat=R.Num\_Bat AND R.Num\_Cou=C.Num\_Cou\\\hspace*{3cm}AND NomC='Solo de Quiberon';
\end{correction}

\item Donner le nom des bateaux qui sont toujours arrivés dans les 5 premiers ŕ la course du 'Solo de Quiberon' \texttt{(2 points)} ;
\begin{correction}
	$\Pi_{NomB}(Bateau\Join Resultat \Join\sigma_{NomC='Solo de Quiberon'}Course)$\\-\\$\Pi_{NomB}(Bateau\Join \sigma_{Score>5}Resultat \Join\sigma_{NomC='Solo de Quiberon'}Course)$\\~\\
	SELECT NomB FROM Bateau B, Resultat R1, Course C1\\WHERE B.Num\_Bat=R1.Num\_Bat AND R1.Num\_Cou=C1.Num\_Cou\\\hspace*{2cm}AND NomC='Solo de Quiberon'\\\hspace*{2cm}
AND Num\_Bat NOT IN\\\hspace*{2.5cm}(SELECT Num\_Bat FROM Resultat R2, Course C2\\\hspace*{2.6cm}WHERE R2.Num\_Cou=C2.Num\_Cou AND C2.NomC='Solo de Quiberon' AND Score>5) ;
\end{correction}



\item[] ~\\\textbf{Donner une requęte en SQL pour les phrases suivantes :}\\

\item Donner pour chaque nom de bateau, le nombre de ses participations ŕ une course \texttt{(1 point)} ;
\begin{correction}
	SELECT NomB, COUNT(*)\\FROM Bateau B, Course C\\WHERE B.Num\_Bat=C.Num\_Bat\\GROUP BY NomB ;
\end{correction}

\item Donner pour chaque sponsor, le nombre de fois que l'un de ses bateaux est arrivé premier. Classer le résultat par ordre décroissant (nombre de courses gagnées) \texttt{(1.5 points)} ;
\begin{correction}
	SELECT Sponsor, COUNT(*) AS GAGNE\\FROM Bateau B, Resultat R, Course C\\WHERE B.Num\_Bat=R.Num\_Bat AND R.Num\_Cou=C.Num\_Cou AND Score=1\\GROUP BY Sponsor\\ORDER BY GAGNE DESC ;
\end{correction}
\end{enumerate}

\section{Concurrence (6 points)}

Soit l'histoire suivante :\\
H = $r_1[x]$ $r_2[x]$ $w_3[y]$ $r_2[y]$ $w_2[y]$ $r_3[z]$ $w_1[y]$ $c_3$ $w_1[z]$ $w_2[z]$ $c_2$ $c_1$

\begin{enumerate}

\item Donner le graphe de sérialisation. H est-elle sérialisable ? \texttt{(1 point)}
\begin{correction}
	$T_3 \rightarrow T_1, T_3 \rightarrow T_1, T_2 \rightarrow T_1$. Il n'y a pas de cycle, dont H est sérialisable.
\end{correction}

\item Donner la liste des conflits de H ; \texttt{(1 point)}
\begin{correction}
	\begin{itemize}
		\item[\textbf{sur x}] aucun conflit ;
		\item[\textbf{sur y}] $w_3[y] r_2[y]$ $w_3[y] w_2[y]$ $r_2[y] w_1[y]$ $w_2[y] w_1[y]$ ;
		\item[\textbf{sur z}] $w_3[z] w_1[z]$ ;
	\end{itemize}
\end{correction}

\item H est-elle recouvrable ? Évite-t-elle les annulations en cascade ? \texttt{(2 points)}
\begin{correction}
	L'écriture $w_3[y]$ est validée en $c_3$ avant que les transactions $T_1$ et $T_2$ ne valident. De męme pour $w_2[y]$. Donc H est recouvrable. Par contre, les lectures sales se font avant le commit $c_1$, donc elle ne garantie pas l'annulation en cascade.
\end{correction}
