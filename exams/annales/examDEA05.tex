\section{Requętes conjonctives (8 pts)}

Pour chacune des requętes suivantes, donnez son expression
en algčbre relationnelle, et, au choix, en rčgles \textsc{Datalog} 
ou en calcul relationnel.

\begin{enumerate}
  \item Titre des films parus en 2005
  \item Titres des films réalisés par Clint Eastwood
  \item Quel acteur (nom et prénom) a joué dans Impitoyable\,?
  \item Quel artiste a joué dans un film dont il est réalisateur\,?
\end{enumerate}

\begin{solution}
  \begin{enumerate}
    \item $ans(x) \leftarrow Film (\_, x, \_, \_, \_)$ 

    \item $ans(x) \leftarrow Film (\_, x, \_, y, \_), Artiste (y, 'Clint',
    'Eastwood', \_)$ 

    \item $ans(n, m) \leftarrow Film (x, 'Impitoyable', \_, \_, \_), Role (x, y), 
      Artiste (y, n, m,  \_) $ 

    \item $ans(n, m) \leftarrow Film (x, \_, \_, y, \_), Role (x, y), 
      Artiste (y, n, m,  \_)$ 
  \end{enumerate}
\end{solution}

\begin{enumerate}
  \item Quels films ne sont pas réalisés par Clint Eastwood\,?

  \item Quel acteur a joué dans un film  qui n'est pas réalisé par

%  \item Quel acteur n'a joué dans {\em aucun} film  réalisé par

  \item Quel acteur a joué dans {\em tous} les films réalisés par 
          Clint Eastwood\,?

  \item (\textsc{Datalog} seulement) Donner les titres de tous les films
          de la série \textit{Harry Potter}, sachant
           que le premier s'intitule \textit{L'école des sorciers}.
\end{enumerate}

\begin{solution}
  \begin{enumerate}
    \item $Eastwood(x) \leftarrow Film (x, \_, \_, y, \_), Artiste (y, 'Clint',
    'Eastwood', \_)$ \\
         $ans(y) \leftarrow Film (x, y, \_, \_), \neg Eastwood (x)$

    \item $ans(n, m) \leftarrow Artiste (x, n, m), Role (f, x, \_, \_), \neg Eastwood (f)$

   \item $A (a, n, m) \leftarrow Artiste (a, n, m), Eastwood (x), \neg Role(x, a)$

   \item $ans(n, m) \leftarrow Artiste (x, n, m), \neg A(x, n, m)$

   \item $A(x) \leftarrow Film (x, 'Ecole\ des\ sorciers')$\\
            $A(x) \leftarrow A(y), Film (x, \_, \_, \_, y)$

  \end{enumerate}
\end{solution}
